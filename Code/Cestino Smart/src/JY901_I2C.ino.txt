#include <Arduino.h>

#include <Wire.h>
#include <JY901.h>
/*
Test on Uno R3.
JY901    UnoR3
SDA <---> SDA
SCL <---> SCL
*/
void setup() 
{
  Serial.begin(9600);
  JY901.StartIIC();
} 

void loop() 
{
  //magnetometer
 
  JY901.GetMag();
  debM1(">MX:");
  deblnM1(JY901.stcMag.h[0]);
  debM1(">MY:");
  deblnM1(JY901.stcMag.h[1]);
  debM1(">MZ:");
  deblnM1(JY901.stcMag.h[2]);
 

  //deblnM1("");
  delay(5);
}
